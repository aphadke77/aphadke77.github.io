Pandas EDA Cheat Sheet

Importing Data

Import Libraries

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

Reading Data

# CSV
df = pd.read_csv('filename.csv')

# Excel
df = pd.read_excel('filename.xlsx', sheet_name='Sheet1')

# JSON
df = pd.read_json('filename.json')

# SQL
from sqlalchemy import create_engine
engine = create_engine('sqlite:///database.db')
df = pd.read_sql('SELECT * FROM table_name', engine)

Data Overview

Basic Information

# First rows
df.head(10)

# DataFrame shape (rows, columns)
df.shape

# Column names
df.columns

# Data types
df.dtypes

# Summary information
df.info()

# Basic statistics
df.describe()

Data Cleaning

Missing Values

# Check for missing values
df.isna().sum()

# Drop rows with any missing values
df_clean = df.dropna()

# Fill missing values with a specific value
df['column'].fillna(0, inplace=True)

# Fill with mean
df['numeric_col'].fillna(df['numeric_col'].mean(), inplace=True)

Duplicates

# Check for duplicate rows
duplicates = df.duplicated().sum()

# Drop duplicates
df_unique = df.drop_duplicates()

Data Exploration

Categorical Data Analysis

# Unique values
df['cat_col'].unique()

# Value counts
df['cat_col'].value_counts()

# Cross tabulation
pd.crosstab(df['cat_col1'], df['cat_col2'])

Numerical Data Analysis

# Basic statistics
df['num_col'].describe()

# Correlation matrix
correlation = df.corr()

# Correlation heatmap
sns.heatmap(correlation, annot=True, cmap='coolwarm')

Grouping and Aggregation

# Group by one column
df.groupby('cat_col')['num_col'].mean()

# Multiple aggregations
df.groupby('cat_col').agg({
    'num_col1': ['min', 'max', 'mean'],
    'num_col2': ['count', 'sum']
})

Data Visualization

Distribution Plots

# Histogram
df['num_col'].hist(bins=30, figsize=(10, 6))

# Box Plot
sns.boxplot(data=df, x='cat_col', y='num_col')

# KDE Plot
sns.kdeplot(data=df, x='num_col')

Relationship Plots

# Scatter Plot
plt.scatter(df['x'], df['y'])

# Scatter Plot with Seaborn
sns.scatterplot(data=df, x='x_col', y='y_col', hue='cat_col')

# Pair Plot
sns.pairplot(df[['col1', 'col2', 'col3']])
